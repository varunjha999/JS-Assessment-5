Q.1 Count Characters

        const countCharacters = (str) => {
            let count = 0;
            let count2 = 0;
            // looping through the items
            for (let i = 0; i < str.length; i++) {

                // check if the character is at that position
                if (str.charAt(i) == "A") {
                    count += 1;
                }
                else if (str.charAt(i) == "D") {
                    count2 += 1;
                }
            }
            return count + "" + count2;
        };
